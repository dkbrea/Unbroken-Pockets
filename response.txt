                    duration: 0.3
                });

                element.addEventListener('mouseenter', () => {
                    gsap.to(cursor, { scale: 1.5, duration: 0.3 });
                    gsap.to(cursorFollower, { scale: 1.5, duration: 0.3 });
                });
                
                element.addEventListener('mouseleave', () => {
                    gsap.to(cursor, { scale: 1, duration: 0.3 });
                    gsap.to(cursorFollower, { scale: 1, duration: 0.3 });
                });
            });

            // Initial animations
            const tl = gsap.timeline();
            
            // Fade in background shapes
            tl.to('.shape-1', { opacity: 0.6, duration: 1.5, ease: 'power2.inOut' }, 0)
              .to('.shape-2', { opacity: 0.4, duration: 1.5, ease: 'power2.inOut' }, 0.2)
              .to('.shape-3', { opacity: 0.5, duration: 1.5, ease: 'power2.inOut' }, 0.4);
            
            // Fade in navigation
            tl.to('.nav', { opacity: 1, duration: 1, ease: 'power2.out' }, 0.5);
            
            // Fade in side nav
            tl.to('.side-nav', { opacity: 1, duration: 1, ease: 'power2.out' }, 0.5);
            
            // Stagger nav links
            tl.to('.nav-link, .auth-buttons button', { 
                opacity: 1, 
                y: 0, 
                duration: 0.8, 
                stagger: 0.1,
                ease: 'power2.out' 
            }, 0.7);
            
            // Fade in hero content
            tl.to('.main-title', { 
                opacity: 1, 
                y: 0, 
                duration: 1,
                ease: 'power2.out' 
            }, 1);
            
            tl.to('.subtitle', { 
                opacity: 1, 
                y: 0, 
                duration: 1,
                ease: 'power2.out' 
            }, 1.2);
            
            tl.to('.image-zoom', { 
                opacity: 1, 
                y: 0, 
                duration: 1,
                ease: 'power2.out' 
            }, 1.4);
            
            // Fade in scroll indicator
            tl.to('.scroll-indicator', { 
                opacity: 1, 
                duration: 1,
                ease: 'power2.out' 
            }, 1.8);

                opacity: 1,

                scrollTrigger: {
                    trigger: '.feature-section',

                    end: 'top 50%',
                    scrub: 1
                }

                opacity: 1,

                scrollTrigger: {
                    trigger: '.feature-section',
                    start: 'top 80%',

                    menuItems.forEach(i => i.classList.remove('active'));
                    item.classList.add('active');
                });
            });

            // Staggered text reveal for section titles
            document.querySelectorAll('.section-title').forEach(title => {
                const text = title.textContent;
                let html = '';
                
                for (let i = 0; i < text.length; i++) {
                    if (text[i] === ' ') {
                        html += ' ';
                    } else {
                        html += `<span style="display:inline-block; opacity: 0; transform: translateY(20px);">${text[i]}</span>`;
                    }
                }
                
                title.innerHTML = html;
                
                ScrollTrigger.create({
                    trigger: title,
                    start: 'top 80%',
                    onEnter: () => {
                        gsap.to(title.querySelectorAll('span'), {
                            opacity: 1,
                            y: 0,
                            duration: 0.05,
                            stagger: 0.02,
                            ease: 'power2.out'
                        });
                    }
                });
            });
            
            // Add hover lift effect to interactive elements
            const liftElements = document.querySelectorAll('.feature-item, .gallery-item');
            
            liftElements.forEach(el => {
                el.addEventListener('mouseenter', () => {
                    gsap.to(el, {

                        y: 0,

                    });
                });

            const revealElements = document.querySelectorAll('.reveal-on-scroll');

                    trigger: el,
                    start: 'top 80%',
                    onEnter: () => {
                        el.classList.add('active');
                    }
                });
            });
        });
